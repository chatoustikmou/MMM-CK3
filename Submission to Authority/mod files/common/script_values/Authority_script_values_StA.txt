clan_government_levies_opinion_scale_positive = 0.00605

realm_authority = {
	value = 0
	every_vassal = {
		limit = { highest_held_title_tier >= tier_county }
		add = submission_weighted
	}
	add = {
		every_held_title = {
			limit = { tier = tier_county }
			add = {
				add = county_control
				if = {
					limit = { prev.highest_held_title_tier = tier_empire }
					max = 100
				}
				else_if = {
					limit = { prev.highest_held_title_tier = tier_kingdom }
					max = 75
				}
				else_if = {
					limit = { prev.highest_held_title_tier = tier_duchy }
					max = 50
				}
				else_if = {
					limit = { prev.highest_held_title_tier = tier_county }
					max = 25
				}					
				add = {
					add = county_opinion
					max = 0
				}
				min = 0
			}
		}
		if = {
			limit = {
				NOT = { sub_realm_size = 0 }
			}
			divide = sub_realm_size
		}
	}
	round = yes
}

character_tyranny = {
	value = tyranny
}

land_share = {
	add = sub_realm_size
	if = {
		limit = {
			NOT = { liege.sub_realm_size = 0 }
		}
		divide = liege.sub_realm_size
	}
	multiply = 100
}

submission = {
	if = {
		limit = { has_government = feudal_government }
		value = feudal_submission
	}
	else_if = {
		limit = { has_government = clan_government }
		value = clan_submission
	}
	else_if = {
		limit = { has_government = tribal_government }
		value = tribal_submission
	}
	else_if = {
		limit = { has_government = theocracy_government }
		value = theocracy_submission
	}
	else_if = {
		limit = { has_government = republic_government }
		value = republic_submission
	}
	else = { value = 0 }
	multiply = {
		add = 100
		add = submissivness
		divide = 100
	}
	max = 100
	min = 0
	add = submission_liberty
	add = submission_councillor_task
	max = 100
	min = 0
}

submission_weighted = {
	value = 0
	add = submission
	multiply = land_share
	divide = 100
}

submissivness = {
	value = 0
	add = submissivness_honor
	add = submissivness_vengefulness
	add = submissivness_dread
	max = 100
	min = -100
}

submissivness_dread = {
	add = liege.dread
	subtract = ai_boldness
	min = {
		if = {
			limit = { submissivness_honor < submissivness_vengefulness }
			add = submissivness_honor
		}
		else = { add = submissivness_vengefulness }
	}
	max = 100
}
